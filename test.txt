proto Ord on T {
	lt: fn(T, T): bool,
	gt: fn(T, T): bool,
	eq: fn(T, T): bool,
}

type Maybe[T] {
	case Some(T),
	case None,
}

type Vec[E] {
	chunk: *mut [E],
	cap: uint,
	readable len: uint,
}
proto MutCollection[E] on T {
	iterate: fn(T): Iter[E],
	push: fn(*mut T, E),
}
